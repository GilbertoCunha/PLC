int i, j, aux, vaux, v[10] = [-3, 4, 7, 21, 43, 6, 14, -33, 9, 0]

:: fswap ::
    vaux = v[i]
    v[i] = v[aux]
    v[aux] = vaux
:;

|| main ||
    write ("Array:        ")
    for (i, vaux) -> v ::
        write (f"{vaux} ")
    :;

    write ("\nSorted array: ")
    for (i=0, i<10, i=i+1) ::
        aux = i
        for (j=i+1, j<10, j=j+1) ::
            if v[j] < v[aux] ::
                aux = fswap
            :;
        :;
        write (f"{v[i]} ")
    :;
    write ("\n")
:;
